def column_number(name: str) -> int:

    result = 0
    while name:
        result += (26 ** (len(name) - 1)) * (ord(name[0]) - 64)

def column_number(name: str) -> int:

    total = 0
    for i, j in enumerate(name[::-1]):
        total += (ord(j) - 64) * (26 ** i)
        
    return total
